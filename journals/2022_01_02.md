- 计算机：(C)
	- 循环控制练习：
		- ```
		  #include "stdio.h"
		  int main()
		  {
		      // i初始0，实际执行次数，j同i循环做处理，k同i循环做处理，n输出行数
		      int i, j, n, k;
		      n = 3;
		      for (i = 0; i < n; i++)//一共循环n次，每次一行
		      {
		          //1到n行第一个v之前分别要写0-n-1个空格
		          for (j = i; j > 0; j--)//j和i同步，i越大j每次输出的空格越多
		          {
		              printf(" ");
		          }
		          //空格后输出第一个v
		          printf("v");
		          //v后写空格,原理和上个相同
		          for (k = i; k < 2 * (n - (i + 1)) - 1 + i; k++)
		          {
		              printf(" ");
		          }
		          //每次循环最后写一个v，除了最后一次
		          if (i < n - 1)
		              printf("v\n");
		      }
		  
		      return 0;
		  }
		  ```
